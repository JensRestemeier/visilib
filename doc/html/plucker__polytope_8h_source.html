209 template<class P>
210 void PluckerPolytope<P>::addEdge(size_t aVertex0, size_t aVertex1, PluckerPolyhedron<P>* aPolyhedron)
211 {
212 size_t min = aVertex0;
213 size_t max = aVertex1;
214 if (aVertex0 > aVertex1)
215 {
216 min = aVertex1;
217 max = aVertex0;
218 }
219 V_ASSERT(aVertex0 != aVertex1);
220 V_ASSERT(MathCombinatorial::haveAtLeastNCommonFacets(aPolyhedron->getFacetsDescription(aVertex0), aPolyhedron->getFacetsDescription(aVertex1)));
221 if (mEdges.find(std::pair<size_t, size_t>(min, max)) == mEdges.end())
222 {
223 mEdges.insert(std::pair<size_t, size_t>(min, max));
224 }
225
226 if (mVertices.find(aVertex0) == mVertices.end())
227 {
228 mVertices.insert(aVertex0);
229 }
230 if (mVertices.find(aVertex1) == mVertices.end())
231 {
232 mVertices.insert(aVertex1);
233 }
234 }

236 template<class P>
237 void PluckerPolytope<P>::getExtremalStabbingLinesBackTo3D(std::vector<std::pair<MathVector3d, MathVector3d>>& aStabbingLines, const MathPlane3d& aPlane1, const MathPlane3d& aPlane2)
238 {
239 aStabbingLines.clear();
240 V_ASSERT(mExtremalStabbingLines.size() > 0);
241 for (size_t i = 0; i < mExtremalStabbingLines.size(); i++)
242 {
243 std::pair<MathVector3d, MathVector3d> r = MathGeometry::getBackTo3D(mExtremalStabbingLines[i], aPlane1, aPlane2);
244 aStabbingLines.push_back(r);
245 }
246
247#if 1
248
249 if (mExtremalStabbingLinesFacets.size() == 0)
250 return;
251 for (size_t i = 0; i < mExtremalStabbingLines.size(); i++)
252 {
253 for (size_t j = i + 1; j < mExtremalStabbingLines.size(); j++)
254 {
255 if (MathCombinatorial::haveAtLeastNCommonFacets(mExtremalStabbingLinesFacets[i], mExtremalStabbingLinesFacets[j], 3))
256 {
257 aStabbingLines.push_back(std::pair<MathVector3d, MathVector3d>( aStabbingLines[i].first, aStabbingLines[j].first));
258 aStabbingLines.push_back(std::pair<MathVector3d, MathVector3d>( aStabbingLines[i].second, aStabbingLines[j].second));
259 }
260 }
261 }
262#endif
263
264 V_ASSERT(aStabbingLines.size() > 0);

333 template<class P> template<class S>
334 void PluckerPolytope<P>::computeExtremalStabbingLines(PluckerPolyhedron<P>* polyhedron, S tolerance)
335 {
336 V_ASSERT(mExtremalStabbingLines.empty());
337 V_ASSERT(!mEdgesIntersectingQuadric.empty());
338
339 for (auto iter = mEdgesIntersectingQuadric.begin(); iter != mEdgesIntersectingQuadric.end(); iter++)
340 {
341 const P& v1 = polyhedron->get(iter->first);
342 const P& v2 = polyhedron->get(iter->second);
343 GeometryPositionType p1 = polyhedron->getQuadricRelativePosition(iter->first);
344 GeometryPositionType p2 = polyhedron->getQuadricRelativePosition(iter->second);
345
346 std::vector<P> p;
347
348 const std::vector<size_t>& facets1 = polyhedron->getFacetsDescription(iter->first);
349 const std::vector<size_t>& facets2 = polyhedron->getFacetsDescription(iter->second);
350 std::vector<size_t> edgeFacets;
351 MathCombinatorial::initFacets(facets1, facets2, polyhedron->getLinesCount(), edgeFacets);
352 edgeFacets.pop_back();
353
354 if (MathGeometry::findPluckerEdgeWithQuadricIntersection<P, S>(v1, v2, p1, p2, p, tolerance))
355 {
356 for (size_t index = 0; index < p.size(); index++)
357 {
358 // V_ASSERT(MathPredicates<P>::isNormalized(p[index]));
359 addExtremalStabbingLine(p[index]);
360
361 mExtremalStabbingLinesFacets.push_back(edgeFacets);
362 }
363 }
364 else
365 {
366 V_ASSERT(MathGeometry::hasPluckerEdgeWithQuadricIntersection(v1, v2, p1, p2, tolerance));
367 }
368 }

391 template<class P> template<class S>
392 bool PluckerPolytope<P>::isValid(PluckerPolyhedron<P>* polyhedron, bool normalization, S tolerance)
393 {
394 bool isValid = true;
395
396 if (mEdgesIntersectingQuadric.size() == 0)
397 {
398 std::cerr << " IsValid error: no stabbing line..." << std::endl;
399 isValid = false;
400 }
401
402 if (mEdges.size() == 0)
403 {
404 std::cerr << " IsValid error: no edge..." << std::endl;
405
406 isValid = false;
407 }
408
409 size_t edgeMax = 2000;
410 if ((size_t)mEdges.size() > edgeMax)
411 {
412 std::cerr << " IsValid error: more than " << edgeMax << " edges..." << std::endl;
413
414 isValid = false;
415 }
416
417 if (MathPredicates::hasPluckerPolytopeIntersectionWithQuadric(this, polyhedron) != ON_BOUNDARY)
418 {
419 std::cerr << " IsValid error: polytope not intersecting the quadric" << std::endl;
420
421 isValid = false;
422 }
423
424 if (hasAllEdgesCollapsed(polyhedron, tolerance))
425 {
426 std::cerr << " IsValid error: hasAllEdgesCollapesd" << std::endl;
427 outputProperties(std::cout);
428
429 isValid = false;
430 }
431
432 for (auto iter = mVertices.begin(); iter != mVertices.end(); iter++)
433 {
434 if (normalization)
435 {
436 if (!MathPredicates::isNormalized(polyhedron->get(*iter), tolerance))
437 {
438 std::cerr << " IsValid error: vertex " << *iter << " not normalized" << std::endl;
439
440 isValid = false;
441 }
442 }
443 std::vector<size_t>& myFacets = polyhedron->getFacetsDescription(*iter);
444
445 for (auto i = myFacets.begin(); i != myFacets.end(); i++)
446 {
447 const P& v1 = polyhedron->get(*iter);
448 const P& v2 = polyhedron->get(*i);
449
450 if (!MathPredicates::getVertexPlaneRelativePosition(v1, v2, tolerance) == ON_BOUNDARY)
451 {
452 std::cerr << " IsValid error: polytope vertex " << *iter << " is not on the facet " << *i << std::endl;
453 std::cerr << " P:" << polyhedron->get(*iter) << std::endl;
454 std::cerr << " Facet:" << polyhedron->get(*i) << std::endl;
455 isValid = false;
456 }
457 }
458 }
459
460 if (!isValid)
461 {
462 outputProperties(std::cout,polyhedron);
463 }
464
465 return isValid;
466 }

468 template<class P> template<class S>
469 void PluckerPolytope<P>::computeEdgesIntersectingQuadric(PluckerPolyhedron<P>* polyhedron, S tolerance)
470 {
471 //Only compute if the function has not been called already
472 if (mEdgesIntersectingQuadric.empty())
473 {
474 //At least one
475 if (mEdges.size() > 0 && MathPredicates::hasPluckerPolytopeIntersectionWithQuadric(this, polyhedron))
476 {
477 V_ASSERT(mEdgesIntersectingQuadric.empty());
478
479 for (auto iter = mEdges.begin(); iter != mEdges.end(); iter++)
480 {
481 const P& v1 = polyhedron->get(iter->first);
482 const P& v2 = polyhedron->get(iter->second);
483 GeometryPositionType p1 = polyhedron->getQuadricRelativePosition(iter->first);
484 GeometryPositionType p2 = polyhedron->getQuadricRelativePosition(iter->second);
485 if (MathGeometry::hasPluckerEdgeWithQuadricIntersection(v1, v2, p1, p2, tolerance))
486 {
487 mEdgesIntersectingQuadric.insert(*iter);
488 }
489 }
490 }
491 else
492 {
493 return;
494 }
495 if (!containsRealLines())
496 return;
497 V_ASSERT(mEdgesIntersectingQuadric.size() > 0);
498 }
