44 inline bool MathCombinatorial::haveAtLeastNCommonFacets(const std::vector<size_t>& aFacetsDescription1, const std::vector<size_t>& aFacetsDescription2, size_t n)
45 {
46 V_ASSERT(std::is_sorted(aFacetsDescription1.begin(), aFacetsDescription1.end()));
47 V_ASSERT(std::is_sorted(aFacetsDescription2.begin(), aFacetsDescription2.end()));
48
49 std::vector<size_t> result(aFacetsDescription1.size());
50 auto iter = std::set_intersection(aFacetsDescription1.begin(), aFacetsDescription1.end(), aFacetsDescription2.begin(), aFacetsDescription2.end(), result.begin());
51 size_t size = iter - result.begin();
52
53 return size >= n;
54 }

61 inline void MathCombinatorial::initFacets(const std::vector<size_t>& aFacetsDescription1, const std::vector<size_t>& aFacetsDescription2, size_t anHyperplane, std::vector<size_t>& result)
62 {
63 V_ASSERT(MathCombinatorial::haveAtLeastNCommonFacets(aFacetsDescription1, aFacetsDescription2));
64 bool requireSorting = anHyperplane <= aFacetsDescription1[aFacetsDescription1.size() - 1]
65 || anHyperplane <= aFacetsDescription2[aFacetsDescription2.size() - 1];
66
67 result.resize(aFacetsDescription1.size());
68 auto iter = std::set_intersection(aFacetsDescription1.begin(), aFacetsDescription1.end(), aFacetsDescription2.begin(), aFacetsDescription2.end(), result.begin());
69 size_t size = iter - result.begin();
70 result.resize(size + 1);
71 result[size] = anHyperplane;
72 if (requireSorting)
73 {
74 std::sort(result.begin(), result.end());
75 }
76 V_ASSERT(std::is_sorted(result.begin(), result.end()));
77 }

81 inline bool MathCombinatorial::hasFacet(const std::vector<size_t> & aFacetsDescription, size_t aFace)
82 {
83 V_ASSERT(std::is_sorted(aFacetsDescription.begin(), aFacetsDescription.end()));
84
85 return std::binary_search(aFacetsDescription.begin(), aFacetsDescription.end(), aFace);;
