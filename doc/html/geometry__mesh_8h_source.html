145  inline void GeometryMesh::computeNormal()
146  {
147  if (mIndices.size() <= 4)
148  {
149  MathVector3f d1 = mVertices[1] - mVertices[0];
150  MathVector3f d2 = mVertices[2] - mVertices[0];
151 
152  mNormals[0] = MathVector3f::cross(d1, d2);
153  mNormals[0].normalize();
154  mNormals[1] = mNormals[0];
155  mNormals[2] = mNormals[0];
156  return;
157  }
158 
159  for (size_t i = 0; i < mVertices.size(); i++)
160  {
161  mNormals[i] = MathVector3f::Zero();
162  }
163  for (size_t i = 0; i < mIndices.size() - 4; i = i + 3)
164  {
165  MathVector3f d1 = mVertices[mIndices[i + 1]] - mVertices[mIndices[i]];
166  MathVector3f d2 = mVertices[mIndices[i + 2]] - mVertices[mIndices[i]];
167 
168  MathVector3f n = MathVector3f::cross(d1, d2);
169  n.normalize();
170  mNormals[mIndices[i]] += n;
171  mNormals[mIndices[i + 1]] += n;
172  mNormals[mIndices[i + 2]] += n;
173  }
174 
175  for (size_t i = 0; i < (size_t)mVertices.size(); i++)
176  {
177  mNormals[i].normalize();
178  }
179  }

192  inline void GeometryMesh::removeDegeneratedTriangles(float threshold)
193  {
194  MathVector3f myNormal;
195  MathVector3f d1, d2;
196  if (mIndices.size() == 0)
197  {
198  std::cout << "Warning: Mesh without triangle" << std::endl;
199  return;
200  }
201 
202  if (mVertices.size() == 0)
203  {
204  std::cout << "Warning: Mesh without vertex" << std::endl;
205  return;
206  }
207 
208  if (mIndices.size() <= 4)
209  {
210  myNormal = MathVector3f::cross(mVertices[1] - mVertices[0], mVertices[2] - mVertices[0]);
211  if (myNormal.isZero(threshold))mIndices.clear();
212  }
213  else
214  {
215  for (size_t i = 0; i < mIndices.size() - 3; i += 3)
216  {
217  V_ASSERT(mIndices[i + 0] < mVertices.size());
218  V_ASSERT(mIndices[i + 1] < mVertices.size());
219  V_ASSERT(mIndices[i + 2] < mVertices.size());
220  myNormal = MathVector3f::cross(mVertices[mIndices[i + 1]] - mVertices[mIndices[i]], mVertices[mIndices[i + 2]] - mVertices[mIndices[i]]);
221  if (myNormal.isZero(threshold))
222  {
223  swapLastIndice(i);
224  i -= 3;
225  }
226  }
227  }
228  }
