136  inline bool SilhouetteProcessor::isPolygonBetweenSourcePlanes(SilhouetteMeshFace * face)
137  {
138  auto iter = mPolygonBetweenSourcePlanesCache.find(face);
139  if (iter != mPolygonBetweenSourcePlanesCache.end())
140  return iter->second;
141 
142  bool inside = true;
143 
144  for (int i = 0; i < 2; i++)
145  {
146  bool vertexOnFrontSide = MathGeometry::hasVertexOnFrontSide(mSource[i]->getPlane(), face);

150  inside = false;

152  }
153  mPolygonBetweenSourcePlanesCache.insert(std::pair<SilhouetteMeshFace*, bool>(face, inside));
154  return inside;

162  inline bool SilhouetteProcessor::isPotentialSilhouetteEdge(SilhouetteMeshFace * face0, SilhouetteMeshFace * face1)
163  {
164  size_t key = getKey(face0, face1);
165  auto iter = mPotentialSilhouetteEdgeCache.find(key);

179  inline bool SilhouetteProcessor::isPotentialSilhouetteEdgeInternal(SilhouetteMeshFace * face0, SilhouetteMeshFace * face1)
180  {
181  if (!isPolygonBetweenSourcePlanes(face0) || !isPolygonBetweenSourcePlanes(face1))
182  {
183  return false;
184  }
185 
186  MathPlane3d plane0 = convert<MathPlane3d>(MathGeometry::computePlane(face0->getVertex(0), face0->getVertex(1), face0->getVertex(2)));
187  MathPlane3d plane1 = convert<MathPlane3d>(MathGeometry::computePlane(face1->getVertex(0), face1->getVertex(1), face1->getVertex(2)));
188 
189  for (int s = 0; s < 2; s++)
190  {
191  if (intersect(*mSource[s], plane0) || intersect(*mSource[s], plane1))
192  {
193  continue;
194  }
195  const MathVector3d& myVertex = mSource[s]->getVertex(0);
196  double dot0 = plane0.dot(myVertex);
197  double dot1 = plane1.dot(myVertex);
198  if (MathArithmetic<double>::isSameSign(dot0, dot1))
199  {
200  return false;
201  }
202  }
203 
204  if (!MathArithmetic<double>::isSameSign(plane0.d, plane1.d))
205  {
206  return true;
207  }
208 
209  if (MathGeometry::hasVertexOutsidePlane(plane1, face0))
210  {
211  return true;
212  }
213 
214  return false;
215  }

217  inline void SilhouetteProcessor::extractSilhouette(size_t geometryId, const std::vector<SilhouetteMeshFace> & meshFaces, bool silhouetteOptimization, std::vector<Silhouette*> & silhouettes)
218  {
219  std::vector<bool> processed;
220  processed.resize(meshFaces.size());
221 
222  for (size_t faceIndex = 0; faceIndex < meshFaces.size(); faceIndex++)
223  {
224  if (processed[faceIndex])
225  continue;
226 
227  std::stack<int> stack;
228  stack.push((int)faceIndex);
229 
230  Silhouette* s = new Silhouette(meshFaces, geometryId);
231  silhouettes.push_back(s);
232 
233  while (!stack.empty())
234  {
235  int myIndex = stack.top();
236  stack.pop();
237  if (processed[myIndex])
238  continue;
239 
240  V_ASSERT(!processed[myIndex]);
241  processed[myIndex] = true;
242  SilhouetteMeshFace* face = const_cast<SilhouetteMeshFace*>(&meshFaces[myIndex]);
243  V_ASSERT(findSilhouette(face) == nullptr);
244 
245  if (mConvexHull != nullptr)
246  {
247  std::vector<bool> hasNeighbours(face->getVertexCount(), true);
248 
249  bool faceIsInsideHull = false;
250  for (int edgeIndex = 0; edgeIndex < face->getVertexCount(); edgeIndex++)
251  {
252  MathVector2i edge = face->getEdge(edgeIndex);
253 
254  if (MathGeometry::isEdgePotentiallyInsideShaft(mConvexHull->getFaces(), face->getVertex(edge.x), face->getVertex(edge.y), false))
255  {
256  faceIsInsideHull = true;
257 
258  if (!silhouetteOptimization
259  || face->getNeighbours(edgeIndex) == -1 || isPotentialSilhouetteEdge(face, const_cast<SilhouetteMeshFace*>(&meshFaces[face->getNeighbours(edgeIndex)])))
260  {
261  s->addEdge(face, edgeIndex,0);
262  hasNeighbours[edgeIndex] = false;
263  }
264  }
265  else
266  {
267  hasNeighbours[edgeIndex] = false;
268  }
269  }
270 
271  if (faceIsInsideHull)
272  {
273  addSilhouette(face, s);
274 
275  if (mDebugger != nullptr)
276  {
277  mDebugger->addRemovedTriangle(face->getVertex(0), face->getVertex(1), face->getVertex(2));
278  }
279 
280  s->addFace(*face);
281 
282  mHelperStatisticCollector->inc(OCCLUDER_TRIANGLE_COUNT);
283 
284  if (silhouetteOptimization)
285  {
286  for (int i = 0; i < face->getVertexCount(); i++)
287  {
288  if (hasNeighbours[i]&&!processed[face->getNeighbours(i)])
289  {
290  stack.push(face->getNeighbours(i));
291  }
292  }
293  }
294 
295  }
296  }
297  }
298  }
299  }
