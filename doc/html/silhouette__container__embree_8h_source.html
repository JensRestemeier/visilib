119 inline void occlusionFilter(const RTCFilterFunctionNArguments* args)
120 {
121 /* avoid crashing when debug visualizations are used */
122 if (args->context == nullptr) return;
123
124 assert(args->N == 1);
125 int* valid = args->valid;
126 RTCHit * hit = (RTCHit*)args->hit;
127
128 /* ignore inactive rays */
129 if (valid[0] != -1) return;
130
131 GeometryIntersectContextMultiHit* context = (GeometryIntersectContextMultiHit*)args->context;
132 GeometryRayMultiHit* ray = context->ray;
133
134 for (unsigned int i = ray->firstHit; i < ray->lastHit; i++)
135 {
136 unsigned slot = i % HIT_LIST_LENGTH;
137 if (ray->hit_geomIDs[slot] == hit->geomID) {
138 valid[0] = 0; return; // ignore duplicate intersections
139 }
140 }
141 /* store hit in hit list */
142 unsigned int slot = ray->lastHit % HIT_LIST_LENGTH;
143 ray->hit_geomIDs[slot] = hit->geomID;
144 ray->lastHit++;
145 if (ray->lastHit - ray->firstHit >= HIT_LIST_LENGTH)
146 ray->firstHit++;
147
148 valid[0] = 0;
149 }
150 inline bool SilhouetteContainerEmbree::intersect(VisibilityRay* aRay)
151 {
152 GeometryRayMultiHit ray;
153 ray.firstHit = 0;
154 ray.lastHit = 0;
155 ray.rayHit.ray.org_x = aRay->org[0]; ray.rayHit.ray.org_y = aRay->org[1]; ray.rayHit.ray.org_z = aRay->org[2];
156 ray.rayHit.ray.dir_x = aRay->dir[0]; ray.rayHit.ray.dir_y = aRay->dir[1]; ray.rayHit.ray.dir_z = aRay->dir[2];
157 ray.rayHit.ray.tnear = aRay->tnear;
158 ray.rayHit.ray.tfar = aRay->tfar;
159 ray.rayHit.ray.mask = -1;
160 ray.rayHit.ray.time = 0;
161 ray.rayHit.ray.id = 0;
162
163 ray.rayHit.hit.geomID = RTC_INVALID_GEOMETRY_ID;
164 GeometryIntersectContextMultiHit context;
165 rtcInitIntersectContext(&context.context);
166
167 while (true)
168 {
169 context.ray = &ray;
170 rtcIntersect1(mScene, &context.context, RTCRayHit_(ray));
171 if (ray.rayHit.hit.geomID == RTC_INVALID_GEOMETRY_ID)
172 break;
173 }
174 for (unsigned int i = ray.firstHit; i < ray.lastHit; i++)
175 {
176 auto h = ids[ray.hit_geomIDs[i]];
177 aRay->addIntersection(h.first, h.second, 0.0);
178 }
179 return ray.lastHit > ray.firstHit;
180 }
