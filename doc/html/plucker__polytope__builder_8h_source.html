69 inline void PluckerPolytopeBuilder<P, S>::addSourcePolygonEdgesAsHyperplanes(PluckerPolyhedron<P>* aPolyhedron, const GeometryConvexPolygon& aPolygon, bool clockWise, const MathVector3d& approximateNormal)
70 {
71 size_t n = aPolygon.getVertexCount();
72 std::vector<MathVector3d> v;
73
74 if (n < 3)
75 {
76 // manage the case of a degenerated polygons
77 v.resize(3);
78 v[1] = aPolygon.getVertex(0);
79
80 if (n == 1)
81 {
82 // the polygon is a single vertex. We initialize two temporary vertices
83 v[0] = v[1] + MathVector3d::X();
84 v[2] = v[1] + MathVector3d::Y();
85 }
86 else if (n == 2)
87 {
88 // the polygon is a single edge. We initialize one temporary vertex
89 v[2] = aPolygon.getVertex(1);
90
91 MathVector3d edge = v[2] - v[1];
92 MathVector3d e1 = MathVector3d::cross(approximateNormal, edge);
93 e1.normalize();
94 e1 *= edge.getNorm();
95
96 v[0] = (v[1] + v[2]) * 0.5 + e1;
97 }
98
99 /* We compute an additional vertex that will be used to add two edges to form a valid polygon, in order to make sure that each vertex of the polygon
100 is at the intersection of two edges. This ensure later one in Plucker space that each extremal lines is at the intersection of four hyperplanes.*/
101
102 MathVector3d edge = v[2] - v[1];
103 MathVector3d e1 = MathVector3d::cross(approximateNormal, edge);
104 e1.normalize();
105 e1 *= edge.getNorm();
106
107 v[0] = (v[1] + v[2]) * 0.5 + e1;
108 }
109 else
110 {
111 // manage the case of a regular convex polygon
112 v = aPolygon.getVertices();
113 }
114 size_t count = v.size();
115
116 // Compute the oriented Plucker line representation of each edge and store it in Polyhedron
117 for (size_t i = 0; i < count; i++)
118 {
119 const MathVector3d& v1 = v[i % count];
120 const MathVector3d& v2 = v[(i + 1) % count];
121
122 P line = clockWise ? P(v1, v2) : P(v2, v1);
123 if (mNormalization)
124 {
125 line = line.getNormalized();
126
127 V_ASSERT((MathPredicates::isNormalized(line, mTolerance)));
128 }
129 aPolyhedron->add(line, ON_BOUNDARY, mNormalization, mTolerance);
130 }
131 }

133 template<class P, class S>
134 inline void PluckerPolytopeBuilder<P, S>::addExtremalStabbingLines(PluckerPolytope<P> * polytope, GeometryConvexPolygon & a, GeometryConvexPolygon & b, PluckerPolyhedron<P> * aPolyhedron)
135 {
136 // There is an extremal stabbing line passing through each vertex of polygon a and each vertex of polygon b.
137 // This procedure initialize the facets of each stabbing line, which is the list of edges incident to the lines in 3D space. In Plucker space, it is list of hyperplanes that meet at
138 // the Plucker point corresponding to the stabbing line
139
140 size_t count1 = a.getVertexCount();
141 size_t count2 = b.getVertexCount();
142
143 size_t mySize1 = count1 > 3 ? count1 : 3;
144
145 std::vector<size_t> myTemp(4);
146
147 int myStabbingNumber = 0;
148
149 // we iterate through all the vertex of polygon a
150 for (size_t i = 0; i < count1; i++)
151 {
152 if (count1 > 2)
153 {
154 // regular case: the first two facets are the edges of the polygon a
155 MathVector2i edges = a.getEdgesOfVertex(i);
156 myTemp[0] = edges.x;
157 myTemp[1] = edges.y;
158 }
159 else
160 {
161 // degenated case (polygon a is a single vertex or an edge): the facets are the additional edges that we have added
162 myTemp[0] = i;
163 myTemp[1] = i + 1;
164 }
165
166 // we iterate through all the vertex of polygon b
167 for (int j = 0; j < count2; j++)
168 {
169 const MathVector3d& v1 = a.getVertex(i);
170 const MathVector3d& v2 = b.getVertex(j);
171 P myStabbingLine(v1, v2);
172 if (mNormalization)
173 {
174 myStabbingLine = myStabbingLine.getNormalized();
175 }
176
177 polytope->addExtremalStabbingLine(myStabbingLine);
178 size_t vertexIndex = aPolyhedron->add(myStabbingLine, ON_BOUNDARY, mNormalization, mTolerance);
179
180 if (count2 > 2)
181 {
182 // regular case: the first two facets are the edges of the polygon b
183 MathVector2i edges = b.getEdgesOfVertex(j);
184
185 myTemp[2] = edges.x + mySize1;
186 myTemp[3] = edges.y + mySize1;
187 }
188 else
189 {
190 // degenated case (polygon b is a single vertex or an edge): the facets are the additional edges that we have added
191 myTemp[2] = j + mySize1;
192 myTemp[3] = j + mySize1 + 1;
193 }
194
195 aPolyhedron->initFacetsDescription(vertexIndex, myTemp);
196
197 myStabbingNumber++;
198 }
199 }
200 }

202 template<class P, class S>
203 inline PluckerPolytope<P>* PluckerPolytopeBuilder<P, S>::build(GeometryConvexPolygon & a, GeometryConvexPolygon & b, PluckerPolyhedron<P> * aPolyhedron)
204 {
205 PluckerPolytope<P>* myPolytope = new PluckerPolytope<P>();
206
207 // The gravity centers are used to compute an approximated normal vector, that will be used when polygons are degenerated
208 MathVector3d ga = MathGeometry::getGravityCenter(a);
209 MathVector3d gb = MathGeometry::getGravityCenter(b);
210
211 addSourcePolygonEdgesAsHyperplanes(aPolyhedron, a, true, gb - ga);
212 addSourcePolygonEdgesAsHyperplanes(aPolyhedron, b, false, ga - gb);
213
214 addExtremalStabbingLines(myPolytope, a, b, aPolyhedron);
215 addEdges(myPolytope, aPolyhedron);
216
217 myPolytope->computeEdgesIntersectingQuadric(aPolyhedron, mTolerance);
218 V_ASSERT(myPolytope->isValid(aPolyhedron, mNormalization, mTolerance));
219
220 return myPolytope;
221 }

223 template<class P, class S>
224 inline void PluckerPolytopeBuilder<P, S>::addEdges(PluckerPolytope<P> * polytope, PluckerPolyhedron<P> * aPolyhedron)
225 {
226 // We iterate through all the extremal stabbing lines, and creates and edge for the vertices that share at least three common facets
227 // The complexity is O(n^2), and more optimized version could be devised but we keep this one for simplicity
228 for (int i = 0; i < aPolyhedron->getLinesCount(); i++)
229 {
230 for (int j = i + 1; j < aPolyhedron->getLinesCount(); j++)
231 {
232 if (MathCombinatorial::haveAtLeastNCommonFacets(aPolyhedron->getFacetsDescription(i), aPolyhedron->getFacetsDescription(j)))
233 {
234 polytope->addEdge(i, j, aPolyhedron);
235 }
236 }
237 }
