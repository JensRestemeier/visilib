171 template<class P>
172 void PluckerPolyhedron<P>::addToFacetsDescription(size_t aVertice, size_t aFace)
173 {
174 V_ASSERT(isValid(aVertice));
175
176 size_t length = mFacetsDescription[aVertice].size();
177
178 mFacetsDescription[aVertice].resize(length + 1);
179 size_t i = length;
180
181 for (; i > 0 && mFacetsDescription[aVertice][i - 1] > aFace; --i)
182 {
183 mFacetsDescription[aVertice][i] = mFacetsDescription[aVertice][i - 1];
184 }
185 mFacetsDescription[aVertice][i] = aFace;
186
187 V_ASSERT(isValid(aVertice));
188 }

191 void PluckerPolyhedron<P>::initFacetsDescription(size_t aVerticeDestination, const std::vector<size_t> & source)
192 {
193 //V_ASSERT(mFacetsDescription.size() == aVerticeDestination);
194
195 // a Source is sortded in place, so it can be modified by the function
196 std::vector<size_t>& facets = getFacetsDescription(aVerticeDestination);
197 for (size_t i = 0; i < source.size(); i++)
198 {
199 facets.push_back(source[i]);
200 }
201 std::sort(facets.begin(), facets.end());
202
203 V_ASSERT(isValid(aVerticeDestination));
204 }

206 template<class P>
207 bool PluckerPolyhedron<P>::checkFacetsDescription(size_t aVerticeDestination, size_t aVerticeSource1, size_t aVerticeSource2, size_t aReplacement)
208 {
209 if (aVerticeDestination >= mFacetsDescription.size())
210 {
211 return false;
212 }
213
214 // sanity checks
215 if (!MathCombinatorial::haveAtLeastNCommonFacets(getFacetsDescription(aVerticeSource1), getFacetsDescription(aVerticeSource2)))
216 return false;
217
218#ifdef FULL_VALIDITY_CHECK
219 if (containsOtherLinesWithSameFacetsDescription(aVerticeDestination))
220 return false;
221#endif
222 if (!isValid(aVerticeDestination))
223 return false;
224 if (!isValid(aVerticeSource1))
225 return false;
226 if (!isValid(aVerticeSource2))
227 return false;
228 if (!MathCombinatorial::haveAtLeastNCommonFacets(getFacetsDescription(aVerticeSource1), getFacetsDescription(aVerticeDestination)))
229 return false;
230 if (!MathCombinatorial::haveAtLeastNCommonFacets(getFacetsDescription(aVerticeSource2), getFacetsDescription(aVerticeDestination)))
231 return false;
232 return true;
233 }

242 template<class P>
243 bool PluckerPolyhedron<P>::containsOtherLinesWithSameFacetsDescription(size_t aVerticeDestination) const
244 {
245 const std::vector<size_t>& source = mFacetsDescription[aVerticeDestination];
246 for (size_t i = 0; i < getLinesCount(); i++)
247 {
248 if (i != aVerticeDestination)
249 {
250 const std::vector<size_t>& facets = mFacetsDescription[i];
251 if (source.size() == facets.size() && std::equal(source.begin(), source.end(), facets.begin()))
252 return true;
253 }
254 }
255 return false;
256 }
