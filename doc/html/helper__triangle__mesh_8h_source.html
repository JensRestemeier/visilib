128 inline void HelperTriangleMesh::computeNormal()
129 {
130 if (mIndices.size() <= 4)
131 {
132 MathVector3f d1 = mVertices[1] - mVertices[0];
133 MathVector3f d2 = mVertices[2] - mVertices[0];
134
135 mNormals[0] = MathVector3f::cross(d1, d2);
136 mNormals[0].normalize();
137 mNormals[1] = mNormals[0];
138 mNormals[2] = mNormals[0];
139 return;
140 }
141
142 for (size_t i = 0; i < mVertices.size(); i++)
143 {
144 mNormals[i] = MathVector3f::Zero();
145 }
146 for (size_t i = 0; i < mIndices.size() - 4; i = i + 3)
147 {
148 MathVector3f d1 = mVertices[mIndices[i + 1]] - mVertices[mIndices[i]];
149 MathVector3f d2 = mVertices[mIndices[i + 2]] - mVertices[mIndices[i]];
150
151 MathVector3f n = MathVector3f::cross(d1, d2);
152 n.normalize();
153 mNormals[mIndices[i]] += n;
154 mNormals[mIndices[i + 1]] += n;
155 mNormals[mIndices[i + 2]] += n;
156 }
157
158 for (size_t i = 0; i < (size_t)mVertices.size(); i++)
159 {
160 mNormals[i].normalize();
161 }
162 }

165 {

175 inline void HelperTriangleMesh::removeDegeneratedTriangles(float threshold)
176 {
177 MathVector3f myNormal;
178 MathVector3f d1, d2;
179 if (mIndices.size() == 0)
180 {
181 std::cout << "Warning: Mesh without triangle" << std::endl;
182 return;
183 }
184
185 if (mVertices.size() == 0)
186 {
187 std::cout << "Warning: Mesh without vertex" << std::endl;
188 return;
189 }
190
191 if (mIndices.size() <= 4)
192 {
193 myNormal = MathVector3f::cross(mVertices[1] - mVertices[0], mVertices[2] - mVertices[0]);
194 if (myNormal.isZero(threshold))mIndices.clear();
195 }
196 else
197 {
198 for (size_t i = 0; i < mIndices.size() - 3; i += 3)
199 {
200 V_ASSERT(mIndices[i + 0] < mVertices.size());
201 V_ASSERT(mIndices[i + 1] < mVertices.size());
202 V_ASSERT(mIndices[i + 2] < mVertices.size());
203 myNormal = MathVector3f::cross(mVertices[mIndices[i + 1]] - mVertices[mIndices[i]], mVertices[mIndices[i + 2]] - mVertices[mIndices[i]]);
204 if (myNormal.isZero(threshold))
205 {
206 swapLastIndice(i);
207 i -= 3;
208 }
209 }
210 }
211 }
