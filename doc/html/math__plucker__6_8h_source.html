165  MathPlucker6<S> getNormalized() const
166  {
167  V_ASSERT(MathArithmetic<S>::isFinite(getDirection()));

172  S myDiv = sqrt(mDirection.getSquaredNorm() + mLocation.getSquaredNorm());
173  V_ASSERT(MathArithmetic<S>::getAbs(myDiv) > MathArithmetic<S>::Tolerance());

184  S getDistance(const MathPlucker6<S>& other) const
185  {
186  S myDiv = MathVector3_<S>::cross(getDirection(),other.getDirection()).getNorm();
187 
188  if (MathArithmetic<S>::getAbs(myDiv) > MathArithmetic<S>::Tolerance())
189  {
190  return MathArithmetic<S>::getAbs(dot(other)) / myDiv;
191  }
192  else
193  {
194  // parallel
195  S l1l1 = 1.0 / getDirection().getSquaredNorm();
196  S l2 = other.getDirection().getNorm();
197 
198  S s = l2 * sqrt(l1l1);
199  MathVector3_<S> l = other.getLocation();
200  l /= s;
201  return MathVector3_<S>::cross(getDirection(),getLocation() - l).getNorm() * l1l1;
202  }
