177  {

193  inline bool VisibilitySilhouetteProcessor::isPotentialSilhouetteEdgeInternal(GeometryMeshFace * face0, GeometryMeshFace * face1)
194  {
195  if (!isPolygonBetweenSourcePlanes(face0) || !isPolygonBetweenSourcePlanes(face1))
196  {
197  return false;
198  }
199 
200  MathPlane3d plane0 = MathGeometry::computePlane(face0->getVertex(0), face0->getVertex(1), face0->getVertex(2));
201  MathPlane3d plane1 = MathGeometry::computePlane(face1->getVertex(0), face1->getVertex(1), face1->getVertex(2));
202 
203  for (int s = 0; s < 2; s++)
204  {
205  if (intersect(*mSource[s], plane0) || intersect(*mSource[s], plane1))
206  {
207  continue;
208  }
209  const MathVector3d& myVertex = mSource[s]->getVertex(0);
210  double dot0 = plane0.dot(myVertex);
211  double dot1 = plane1.dot(myVertex);
212  if (MathArithmetic<double>::isSameSign(dot0, dot1))
213  {
214  return false;
215  }
216  }
217 
218  if (!MathArithmetic<double>::isSameSign(plane0.d, plane1.d))
219  {
220  return true;
221  }
222 
223  if (MathGeometry::hasVertexOutsidePlane(plane1, face0))
224  {
225  return true;
226  }
227 
228  return false;
229  }

231  inline VisibilitySilhouette* VisibilitySilhouetteProcessor::extractSilhouette(GeometryMeshFace * aFace, bool silhouetteOptimization)
232  {
233  std::stack<GeometryMeshFace*> stack;
234 
235  stack.push(aFace);
236 
237  VisibilitySilhouette* s = findSilhouette(aFace);
238  if (s == nullptr)
239  {
240  s = new VisibilitySilhouette();
241  }
242 
243  while (!stack.empty())
244  {
245  GeometryMeshFace* face = stack.top();
246  stack.pop();
247 
248  //V_ASSERT(mConvexHull != nullptr);
249 
250  VisibilitySilhouette* myT = findSilhouette(face);
251  if (myT == nullptr)
252  {
253  //TODO clip : manage to restore before succcessive query
254  //face->getPolygon().clipWithGardBand(mSource[0]->getPlane(),MathArithmetic::GuardBandClipping());
255  //face->getPolygon().clipWithGardBand(mSource[1]->getPlane(), MathArithmetic::GuardBandClipping());
256 
257  addSilhouette(face, s);
258 
259  if (mDebugger != nullptr)
260  mDebugger->get()->addRemovedTriangle(face->getVertex(0), face->getVertex(1), face->getVertex(2));
261 
262  mHelperStatisticCollector->inc(OCCLUDER_TRIANGLE_COUNT);
263 
264  if (mConvexHull != nullptr)
265  {
266  std::vector<bool> hasNeighbours(face->getVertexCount(), true);
267 
268  for (int edgeIndex = 0; edgeIndex < face->getVertexCount(); edgeIndex++)
269  {
270  MathVector2i edge = face->getEdge(edgeIndex);
271 
272  if (MathGeometry::isEdgePotentiallyInsideShaft(mConvexHull->getFaces(), face->getVertex(edge.x), face->getVertex(edge.y), false))
273  {
274  if (!silhouetteOptimization || face->getNeighbours(edgeIndex) == nullptr || isPotentialSilhouetteEdge(face, face->getNeighbours(edgeIndex)))
275  {
276  s->add(face, edgeIndex);
277  hasNeighbours[edgeIndex] = false;
278  }
279  }
280  else
281  {
282  hasNeighbours[edgeIndex] = false;
283  }
284  }
285 
286  if (silhouetteOptimization)
287  {
288  for (int i = 0; i < face->getVertexCount(); i++)
289  {
290  if (hasNeighbours[i])
291  {
292  stack.push(face->getNeighbours(i));
293  }
294  }
295  }
296  }
297  }
298  else
299  {
300  V_ASSERT(s == myT);
301  }
302  }
303 
304  return s;
305  }
