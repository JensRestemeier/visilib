107  inline bool GeometryScene::intersect(Ray* aRay)
108  {
109  GeometryRay ray(*aRay);
110 
111  TriangleMeshInformation mesh;
112 
113  for (size_t index = 0; index < mGeometryIds.size(); index++)
114  {
115  size_t id = mGeometryIds[index];
116 
117  if (MathGeometry::hitsBoundingBox(ray, getBoundingBox(id)))
118  {
119  getTriangleMeshInformation(id, mesh);
120  const int* myIndices = mesh.indexArray;
121  const MathVector3f* myVertices = (MathVector3f*)mesh.vertexArray;
122 
123  for (size_t i = 0; i < mesh.indexCount / 3; i++)
124  {
125  if (MathGeometry::hitsTriangle<float>(ray, myVertices[myIndices[i * 3]], myVertices[myIndices[i * 3 + 1]], myVertices[myIndices[i * 3 + 2]]))
126  {
127  aRay->primitiveId = i;
128  aRay->geometryId = id;
129  return true;
130  }
131  }
132  }
133  }
134  return false;
135  }
