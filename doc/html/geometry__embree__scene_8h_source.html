94  inline bool GeometryEmbreeScene::intersect(Ray* aRay)
95  {
96  RTCRay ray;
97  ray.org_x = aRay->org[0]; ray.org_y = aRay->org[1]; ray.org_z = aRay->org[2];
98  ray.dir_x = aRay->org[0]; ray.dir_y = aRay->org[1]; ray.dir_z = aRay->org[2];
99  ray.tnear = aRay->tnear;
100  ray.tfar = aRay->tfar;
101  ray.mask = -1;
102  ray.time = 0;
103 
104  RTCIntersectContext context;
105  rtcInitIntersectContext(&context);
106 
107  RTCRayHit rayHit;
108  rayHit.ray = ray;
109  rayHit.hit.geomID = RTC_INVALID_GEOMETRY_ID;
110 
111  rtcIntersect1(mScene, &context, &rayHit);
112 
113  if (rayHit.hit.geomID != RTC_INVALID_GEOMETRY_ID)
114  {
115  aRay->geometryId = rayHit.hit.geomID;
116  aRay->primitiveId = rayHit.hit.primID;
117  return true;
118  }
119 
120  //Since Embree can miss some intersection around edges, we always use the robust watertight intersection before reporting visibility
121  return GeometryScene::intersect(aRay);
122  }

124  inline void GeometryEmbreeScene::prepare()
125  {
126  GeometryScene::prepare();
127 
128  for (int i = 0; i < mMeshArray.size(); i++)
129  {
130  int id = i;
131  TriangleMeshInformation triangleMesh;
132  getTriangleMeshInformation(id, triangleMesh);
133  RTCGeometry mesh = rtcNewGeometry(mDevice, RTC_GEOMETRY_TYPE_TRIANGLE);
134  rtcSetSharedGeometryBuffer(mesh, RTC_BUFFER_TYPE_VERTEX,0, RTC_FORMAT_FLOAT3, triangleMesh.vertexArray, 0, sizeof(float),triangleMesh.vertexCount);
135  rtcSetSharedGeometryBuffer(mesh, RTC_BUFFER_TYPE_INDEX,0, RTC_FORMAT_UINT3, triangleMesh.indexArray, 0, sizeof(int),triangleMesh.indexCount/3);
136  rtcCommitGeometry(mesh);
137  rtcAttachGeometry(mScene, mesh);
138  }
139 
140 
141  rtcCommitScene(mScene);
142  }
