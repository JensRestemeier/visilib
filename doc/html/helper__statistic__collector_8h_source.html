78 {
79 mCounts.resize(COUNTER_LAST, 0);

81 mTimerIsRunning.resize(TIMER_LAST, false);

84 void inc(CounterType counter)

94 void incrementTime(TimerType counter, double time)
95 {
96 mTimers[counter] += time / mTimerFrequency;
97 }

99 void setTimerIsRunning(TimerType timer, bool isRunning)
100 {
101 V_ASSERT(mTimerIsRunning[timer] != isRunning);
102
103 mTimerIsRunning[timer] = isRunning;

106 double getUnkownTime()
107 {
108 double known = 0;
109 known += mTimers[RAY_INTERSECTION];
110 known += mTimers[SILHOUETTE_PROCESSING];
111 known += mTimers[POLYTOPE_SPLIT];
112 known += mTimers[POLYTOPE_BUILD];
113 known += mTimers[STABBING_LINE_EXTRACTION];
114 known += mTimers[OCCLUDER_TREATMENT];
115
116 return mTimers[VISIBILITY_QUERY] - known;
117 }

164 class HelperScopedTimer
165 {
166 public:
167 HelperScopedTimer(HelperStatisticCollector* aCollector, TimerType aTimer)
168 : mTimer(aTimer),
169 mCollector(aCollector)
170 {
171#ifdef WIN32
172 QueryPerformanceCounter((LARGE_INTEGER*)mStartTime);
173#else
174 mStartTime = clock();
175#endif
176 mCollector->setTimerIsRunning(mTimer, true);
177 }
178
179 ~HelperScopedTimer()
180 {
181#ifdef WIN32
182 unsigned long ellapsedTime[2];
183 QueryPerformanceCounter((LARGE_INTEGER*)ellapsedTime);
184 ellapsedTime[0] -= mStartTime[0];
185 mCollector->incrementTime(mTimer, ellapsedTime[0]);
186#else
187 mCollector->incrementTime(mTimer, (double) (clock()- mStartTime));
188#endif
189
190 mCollector->setTimerIsRunning(mTimer, false);
191 }
192
193 private:
194#ifdef WIN32
195 unsigned long mStartTime[2];
196#else
197 clock_t mStartTime;
198#endif
199 TimerType mTimer;
200 HelperStatisticCollector* mCollector;
201 };
