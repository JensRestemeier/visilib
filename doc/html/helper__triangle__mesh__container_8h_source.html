114 inline GeometryDiscreteMeshDescription* HelperTriangleMeshContainer::createTriangleMeshDescription(size_t geometryId) const
115 {
116 HelperTriangleMesh* mesh = mMeshArray[geometryId];
117 GeometryTriangleMeshDescription* info = new GeometryTriangleMeshDescription;
118 info->vertexCount = mesh->getVertexCount();
119 info->faceCount= mesh->getIndiceCount() / 3;
120 info->indexArray = &mesh->getIndices()[0];
121 info->vertexArray = (float*)& mesh->getVertices()[0];
122
123 info->normalArray = mesh->getNormals().empty() ? nullptr : (float*)& mesh->getNormals()[0];
