87  inline std::vector<GeometryMeshFace>* GeometrySceneData::getFaces(Scene* scene, size_t geometryId)
88  {
89  std::vector<GeometryMeshFace>* myFaces;
90  auto iter = mMeshFacesCache.find(geometryId);
91  if (iter != mMeshFacesCache.end())
92  {
93  myFaces = iter->second;
94  }
95  else
96  {
97  TriangleMeshInformation mesh;
98  scene->getTriangleMeshInformation(geometryId, mesh);
99 
100  myFaces = new std::vector<GeometryMeshFace>();
101  extractFaces(&mesh, *myFaces);
102  mMeshFacesCache.insert(std::pair<size_t, std::vector<GeometryMeshFace>*>(geometryId, myFaces));
103  }
104  return myFaces;
105  }

120  inline void GeometrySceneData::extractFaces(TriangleMeshInformation* mesh, std::vector<GeometryMeshFace>& aFaces)
121  {
122  const int* myIndices = mesh->indexArray;
123 
124  size_t myFaceNumber = mesh->indexCount / 3;
125 
126  aFaces.clear();
127  aFaces.resize(myFaceNumber);
128 
129  for (size_t i = 0; i < myFaceNumber; i++)
130  {
131  aFaces[i].setGeometry(mesh, i);
132  }
133  std::unordered_map<size_t, MathVector2i> myEdgesMap;
134 
135  for (size_t myTriangle = 0; myTriangle < myFaceNumber; myTriangle++)
136  {
137  for (size_t myEdge = 0; myEdge < 3; myEdge++)
138  {
139  size_t myBegin = myIndices[myTriangle * 3 + myEdge];
140  size_t myEnd = myIndices[myTriangle * 3 + (myEdge + 1) % 3];
141 
142  const size_t key = getKey(mesh->indexCount, myBegin, myEnd);
143  auto iter = myEdgesMap.find(key);
144  if (iter == myEdgesMap.end())
145  {
146  aFaces[myTriangle].setNeighbour(myEdge, nullptr);
147  myEdgesMap.insert(std::pair<size_t, MathVector2i>(key, MathVector2i(myTriangle, myEdge)));
148  }
149  else
150  {
151  size_t myNeighborTriangle = iter->second.x;
152  size_t myNeighborEdge = iter->second.y;
153 
154  myEdgesMap.erase(iter);
155 
156  aFaces[myTriangle].setNeighbour(myEdge, &aFaces[myNeighborTriangle]);
157  aFaces[myNeighborTriangle].setNeighbour(myNeighborEdge, &aFaces[myTriangle]);
158  }
159  }
160  }
161  }
