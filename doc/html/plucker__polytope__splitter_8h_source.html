54 template<class P, class S>
55 inline GeometryPositionType PluckerPolytopeSplitter<P, S>::split(PluckerPolyhedron<P>* aPolyhedron, const P& aPlane, PluckerPolytope<P>* aPolytope, PluckerPolytope<P>* aLeft, PluckerPolytope<P>* aRight, size_t aPlaneID, bool normalization, S tolerance)
56 {
57 bool hasPointOnTheLeft = false;
58 bool hasPointOnTheRight = false;
59
60 std::vector<size_t> myWait;
61 std::unordered_set<size_t> myVertices;
62 std::vector<size_t> myQueryList;
63
64 std::unordered_map<size_t, int> myStatus;
65 std::unordered_map<size_t, S> myArrayOfPlanePosition;
66
67 V_ASSERT(aPolytope->getEdgeCount() > 0);
68
69 size_t edge[2];
70
71
72 // Step 1 - edge classification
73
74 // We iterate through all the edges to inspect the relative position of the two myVertices of the edge with regards to the splitting hyperplane
75
76 for (auto iter = aPolytope->getEdges().begin(); iter != aPolytope->getEdges().end(); iter++)
77 {
78 edge[0] = iter->first;
79 edge[1] = iter->second;
80
81 for (size_t i = 0; i < 2; i++)
82 {
83 size_t a = edge[i];
84
85 if (myVertices.find(a) == myVertices.end())
86 {
87 myVertices.insert(a);
88
89 S result = aPlane.dot(aPolyhedron->get(a));
90 GeometryPositionType position = MathPredicates::getVertexPlaneRelativePosition(aPlane, aPolyhedron->get(a), tolerance);
91
92 myArrayOfPlanePosition.insert(std::pair<size_t, S>(a, result));
93
94 if (position == ON_NEGATIVE_SIDE)
95 {
96 myStatus.insert(std::pair<size_t, int>(a, -1));
97 hasPointOnTheLeft = true;
98 }
99 else if (position == ON_POSITIVE_SIDE)
100 {
101 myStatus.insert(std::pair<size_t, int>(a, 1));
102 hasPointOnTheRight = true;
103 }
104 else
105 {
106 V_ASSERT(position == ON_BOUNDARY);
107 myStatus.insert(std::pair<size_t, int>(a, 0));
108 myWait.push_back(a);
109 myQueryList.push_back(a);
110 }
111 }
112 }
113 }
114
115 // Step 2 - Treat the special cases, when the hyperplane does not split with the polytope
116
117 if (!hasPointOnTheLeft && !hasPointOnTheRight)
118 {
119 // Step 2 - Case 1 - the hyperplanes contains all the myVertices of the polytope
120 // - no split is required, the hyperplane is added to the facets description of the myVertices contained in the hyperplane.
121
122 for (auto iter = aPolytope->getEdges().begin(); iter != aPolytope->getEdges().end(); iter++)
123 {
124 size_t myI1 = iter->first;
125 size_t myI2 = iter->second;
126 aLeft->addEdge(myI1, myI2, aPolyhedron);
127 aRight->addEdge(myI1, myI2, aPolyhedron);
128 }
129 V_ASSERT(aLeft->getEdgeCount() > 0);
130 V_ASSERT(aRight->getEdgeCount() > 0);
131 for (size_t index = 0; index < myWait.size(); index++)
132 {
133 if (!MathCombinatorial::hasFacet(aPolyhedron->getFacetsDescription(myWait[index]), aPlaneID))
134 {
135 aPolyhedron->addToFacetsDescription(myWait[index], aPlaneID);
136 }
137 }
138 return ON_BOUNDARY;
139 }
140 if (!hasPointOnTheLeft)
141 {
142 // Step 2 - Case 2 - the polytope is entierly on the positive side. Nothing to do.
143 return ON_POSITIVE_SIDE;
144 }
145 else if (!hasPointOnTheRight)
146 {
147 // Step 2 - Case 3 - the polytope is on entierly the negative side. Nothing to do.
148 return ON_NEGATIVE_SIDE;
149 }
150
151
152 // Step 3 - P processing
153 // We are in the generic case, where the hyperplane split the polytope in two convex polytopes.
154 // We first process all the myVertices and creates the new myVertices if required
155 //
156
157 // The hyperplane is added to the facets description of the myVertices contained in the hyperplane.
158 for (size_t index = 0; index < myWait.size(); index++)
159 {
160 if (!MathCombinatorial::hasFacet(aPolyhedron->getFacetsDescription(myWait[index]), aPlaneID))
161 {
162 aPolyhedron->addToFacetsDescription(myWait[index], aPlaneID);
163 }
164 }
165
166 // We iterate though all the edges to determine the operation required for each edge
167 size_t initialNumberOfVertices = aPolyhedron->getLinesCount();
168
169 for (auto iter = aPolytope->getEdges().begin(); iter != aPolytope->getEdges().end(); iter++)
170 {
171 size_t myI1 = iter->first;
172 size_t myI2 = iter->second;
173
174 int sum = myStatus[myI1] + myStatus[myI2];
175
176 // myStatus[myI1] | + + 0 | + 0 - | 0 - - | position of vertex 0
177 // myStatus[myI2] | + 0 + | - 0 + | - 0 - | position of vertex 1
178 // | ======= |=======|========== |
179 // sum | 2 1 1 | 0 0 0 | -1 -1 -2 | the sum
180 // operation | add to right | split | add to left | the operation to perform
181
182 if (sum < 0) // add edge to left polytope
183 {
184 aLeft->addEdge(myI1, myI2, aPolyhedron);
185 }
186 else if (sum > 0)// add edge to right polytope
187 {
188 aRight->addEdge(myI1, myI2, aPolyhedron);
189 }
190 else // split edge
191 {
192 if (myStatus[myI1] != 0)
193 {
194 // the edge crosses the hyperplane: a new vertex is created at the intersection the edge and the hyperplane
195
196 if (myStatus[myI1] > 0)
197 {
198 size_t temp = myI1;
199 myI1 = myI2;
200 myI2 = temp;
201 }
202
203 std::vector<size_t> myFacets;
204 MathCombinatorial::initFacets(aPolyhedron->getFacetsDescription(myI1),
205 aPolyhedron->getFacetsDescription(myI2),
206 aPlaneID,
207 myFacets);
208
209
210 bool hasVertex = false;
211 for (size_t test = initialNumberOfVertices; test < aPolyhedron->getLinesCount() && !hasVertex; test++)
212 {
213 const std::vector<size_t> existingFacets = aPolyhedron->getFacetsDescription(test);
214 if (existingFacets.size() == myFacets.size() && std::equal(existingFacets.begin(), existingFacets.end(), myFacets.begin()))
215 hasVertex = true;
216 }
217
218 if (hasVertex)
219 continue;
220
221 // Compute vertex position, at the intersection of the edge and the splitting hyperplane
222 P myIntersection = MathGeometry::interpolate(myArrayOfPlanePosition[myI1], myArrayOfPlanePosition[myI2], aPolyhedron->get(myI1), aPolyhedron->get(myI2), tolerance);
223 if (normalization)
224 {
225 myIntersection = myIntersection.getNormalized();
226 }
227
228 size_t vertexIndex = aPolyhedron->add(myIntersection, MathPredicates::getQuadricRelativePosition(myIntersection, tolerance), normalization, tolerance);
229
230 // Prepare for new edges creation
231 myQueryList.push_back(vertexIndex);
232
233 aPolyhedron->initFacetsDescription(vertexIndex, myFacets);
234
235 V_ASSERT(aPolyhedron->checkFacetsDescription(vertexIndex, myI1, myI2, aPlaneID));
236
237 // Add the edges to the left and to the right
238 aLeft->addEdge(myI1, vertexIndex, aPolyhedron);
239 aRight->addEdge(vertexIndex, myI2, aPolyhedron);
240 }
241 else
242 {
243 // the myVertices are on the hyperplane, we do not have to split the edge
244
245 aLeft->addEdge(myI1, myI2, aPolyhedron);
246 aRight->addEdge(myI1, myI2, aPolyhedron);
247 }
248 }
249 }
250
251 //** @brief
252 // Step 4 - Creation of the edges of the polytope skeleton
253
254 // We iterate through all the new extremal stabbing lines, and creates and edge for the myVertices that share at least three common facets and if the edge will not be
255 // degenerated (length 0)
256 for (size_t m = 0; m < myQueryList.size(); m++)
257 {
258 for (size_t n = m + 1; n < myQueryList.size(); n++)
259 {
260 size_t Qm = myQueryList[m];
261 size_t Qn = myQueryList[n];
262
263 const std::vector<size_t>& facetsQm = aPolyhedron->getFacetsDescription(Qm);
264 const std::vector<size_t>& facetsQn = aPolyhedron->getFacetsDescription(Qn);
265
266 if (MathCombinatorial::haveAtLeastNCommonFacets(facetsQm, facetsQn))
267 {
268 V_ASSERT(Qm != Qn);
269 if (!MathPredicates::isEdgeCollapsed(aPolyhedron->get(Qn), aPolyhedron->get(Qm), tolerance))
270 {
271 V_ASSERT(facetsQm.size() != facetsQn.size() || !std::equal(facetsQm.begin(), facetsQm.end(), facetsQn.begin()));
272
273 aLeft->addEdge(Qm, Qn, aPolyhedron);
274 aRight->addEdge(Qm, Qn, aPolyhedron);
275 }
276 }
277 }
278 }
279
280 V_ASSERT(aLeft->getEdgeCount() > 0);
281 V_ASSERT(aRight->getEdgeCount() > 0);
282 // aLeft->V_ASSERTIsValid(aPolyhedron, normalization);
283 // aRight->V_ASSERTIsValid(aPolyhedron, normalization);
284 return ON_BOUNDARY;
285 }
286}
