118  inline void HelperSyntheticMeshBuilder::subdivide(std::vector<int>& indices, std::vector<MathVector3f>& vertices, SubdivisionPatternType pattern)
119  {
120  size_t size = indices.size();
121 
122  std::vector<std::pair<size_t, size_t>> s;
123  s.push_back(std::pair<size_t, size_t>(0, 1));
124  s.push_back(std::pair<size_t, size_t>(1, 2));
125  s.push_back(std::pair<size_t, size_t>(0, 2));
126 
127  std::unordered_map<size_t, size_t> myEdgesMap;
128 
129  size_t i[6];
130  indices.resize(size * 4);
131  size_t lastIndex = size * 4 - 1;
132 
133  for (int index = (int)size - 3; index >= 0; index -= 3)
134  {
135  i[0] = indices[index + 0];
136  i[1] = indices[index + 1];
137  i[2] = indices[index + 2];
138  i[3] = 0;
139  i[4] = 0;
140  i[5] = 0;
141 
142  for (size_t split = 0; split < 3; split++)
143  {
144  size_t a = s[split].first;
145  size_t b = s[split].second;
146  size_t key = getKey(size, i[a], i[b]);
147  auto iter = myEdgesMap.find(key);
148  if (iter != myEdgesMap.end())
149  {
150  i[split + 3] = iter->second;
151  }
152  else
153  {
154  i[split + 3] = vertices.size();
155  MathVector3f newVertex = (vertices[i[a]] + vertices[i[b]]) * 0.5;
156  vertices.push_back(newVertex);
157  myEdgesMap.insert(std::pair<size_t, size_t>(key, i[split + 3]));
158  }
159  }
160 
161  if (pattern == Regular)
162  {
163  indices[lastIndex--] = (int)i[0]; indices[lastIndex--] = (int)i[3]; indices[lastIndex--] = (int)i[5];
164  indices[lastIndex--] = (int)i[3]; indices[lastIndex--] = (int)i[1]; indices[lastIndex--] = (int)i[4];
165  indices[lastIndex--] = (int)i[5]; indices[lastIndex--] = (int)i[4]; indices[lastIndex--] = (int)i[2];
166  indices[lastIndex--] = (int)i[3]; indices[lastIndex--] = (int)i[4]; indices[lastIndex--] = (int)i[5];
167  }
168  /* else if (pattern == OneInside)
169  {
170  indices[index + 0] = i[3]; indices[index + 1] = i[4]; indices[index + 2] = i[5];
171  }
172  else if (pattern == ThreeOutside)
173  {
174  indices[index + 0] = i[0]; indices[index + 1] = i[3]; indices[index + 2] = i[5];
175  indices.push_back(i[3]); indices.push_back(i[1]); indices.push_back(i[4]);
176  indices.push_back(i[5]); indices.push_back(i[4]); indices.push_back(i[2]);
177  }*/
178  }
179  }

275  inline void HelperSyntheticMeshBuilder::generateRegularPolygon(std::vector<MathVector3f> & vertices, size_t vertexCount)
276  {
277  if (vertexCount == 1)
278  {
279  vertices.push_back(MathVector3f::Zero());
280  }
281  else if (vertexCount == 2)
282  {
283  vertices.push_back(MathVector3f(-1, 0, 0));
284  vertices.push_back(MathVector3f(1, 0, 0));
285  }
286  else
287  {
288  const float PI = 3.14159265358979323846f;
289 
290  float theta = 0;
291  float dt = -2 * PI / (float)vertexCount;
292 
293  for (size_t i = 0; i < vertexCount; i++)
294  {
295  vertices.push_back(MathVector3f(cos(theta), sin(theta), 0));
296 
297  theta += dt;
298  }
299  }

302  inline void HelperSyntheticMeshBuilder::removeFaces(GeometryMesh * mesh, size_t count)
303  {
304  std::vector<int>& indices = mesh->getIndices();
305 
306  for (size_t i = 0; i < count; i++)
307  {
308  size_t face = rand() % (indices.size() / 3);
309 
310  indices[face * 3 + 0] = indices[indices.size() - 3];
311  indices[face * 3 + 1] = indices[indices.size() - 2];
312  indices[face * 3 + 2] = indices[indices.size() - 1];
313  indices.resize(indices.size() - 3);
314  }

317  inline void HelperSyntheticMeshBuilder::addRandomness(std::vector<MathVector3f> & vertices, float randomness)
318  {

331  inline void HelperSyntheticMeshBuilder::rescaleToUnitBox(GeometryScene * scene)
332  {
333  auto myMeshArray = scene->getMeshArray();
334 
335  std::vector<MathVector3f> myPoints;
336  for (auto iter = myMeshArray.begin(); iter != myMeshArray.end(); iter++)
337  {
338  (*iter)->computeBoundingBox();
339  myPoints.push_back((*iter)->getBoundingBox().getMin());
340  myPoints.push_back((*iter)->getBoundingBox().getMax());
341  }
342 
343  MathVector3f myMin;
344  MathVector3f myMax;
345 
346  MathArithmetic<float>::getMinMax(myPoints, myMin, myMax);
347 
348  MathVector3f myMiddle = (myMin + myMax) * (-0.5);
349  MathVector3f myExtent = myMax - myMin;
350  float m = myExtent.x;
351  m = m > myExtent.y ? m : myExtent.y;
352  m = m > myExtent.z ? m : myExtent.z;
353  m = 1.0f / m;
354 
355  for (auto iter = myMeshArray.begin(); iter != myMeshArray.end(); iter++)
356  {
357  translate((*iter), myMiddle);
358  scale((*iter), m);
359  }
360  }
