225 template<class P, class S>
226 bool VisibilityExactQuery_<P, S>::createInitialPolygons(const float* vertices0, size_t numVertices0, const float* vertices1, size_t numVertices1, bool)
227 {
228 GeometryConvexPolygon myQuery0(vertices0, numVertices0);
229 GeometryConvexPolygon myQuery1(vertices1, numVertices1);
230
231 MathVector3d g1 = MathGeometry::getGravityCenter(myQuery0);
232 MathVector3d g2 = MathGeometry::getGravityCenter(myQuery1);
233
234 MathVector3d myN = g2 - g1;
235
236 if (myQuery0.getVertexCount() < 3 || myQuery1.getVertexCount() < 3)
237 {
238
239 if (myQuery0.getVertexCount() == 2)
240 {
241 myQuery0.setPlane(MathGeometry::computePlaneFromApproximateNormal(myQuery0, myN));
242 }
243 else if (myQuery0.getVertexCount() == 1)
244 {
245 MathVector3d myNormal = myN;
246 myNormal.normalize();
247 myQuery0.setPlane(MathPlane3d(myNormal.x, myNormal.y, myNormal.z, -myNormal.dot(myQuery0.getVertex(0))));
248 }
249
250 if (myQuery1.getVertexCount() == 2)
251 {
252 myN.x = -myN.x; myN.y = -myN.y; myN.z = -myN.z;
253 myQuery1.setPlane(MathGeometry::computePlaneFromApproximateNormal(myQuery1, myN));
254 }
255 else if (myQuery1.getVertexCount() == 1)
256 {
257 myN.x = -myN.x; myN.y = -myN.y; myN.z = -myN.z;
258 MathVector3d myNormal = myN;
259 myNormal.normalize();
260 myQuery1.setPlane(MathPlane3d(myNormal.x, myNormal.y, myNormal.z, -myNormal.dot(myQuery1.getVertex(0))));
261 }
262 }
263 setApproximateNormal(myN);
264
265 if (!MathGeometry::clipWithGardBand(myQuery0, myQuery1.getPlane(), MathArithmetic<double>::GuardBandClipping()))
266 {
267 std::cout << "Clipping failure" << std::endl;
268 return false;
269 }
270 V_ASSERT(myQuery0.isValid());
271
272 if (!MathGeometry::clipWithGardBand(myQuery1, myQuery0.getPlane(), MathArithmetic<double>::GuardBandClipping()))
273 {
274 std::cout << "Clipping failure" << std::endl;
275 return false;
276 }
277 V_ASSERT(myQuery1.isValid());
278
279 V_ASSERT(MathPredicates::getRelativePosition(myQuery0.getVertices(), myQuery1.getPlane()) == ON_POSITIVE_SIDE);
280 V_ASSERT(MathPredicates::getRelativePosition(myQuery1.getVertices(), myQuery0.getPlane()) == ON_POSITIVE_SIDE);
281
282 mQueryPolygon[0] = new GeometryConvexPolygon(myQuery0);
283 mQueryPolygon[1] = new GeometryConvexPolygon(myQuery1);
284
285 return true;
286 }

288 template<class P, class S>
289 VisibilityResult VisibilityExactQuery_<P, S>::arePolygonsVisible(const float* vertices0, size_t numVertices0, const float* vertices1, size_t numVertices1)
290 {
291 VisibilityResult result;
292
293 HelperScopedTimer timer(&mStatistic, VISIBILITY_QUERY);
294
295 if (mDebugger != nullptr)
296 {
297 mDebugger->clear();
298 mSilhouetteProcessor->attachVisualisationDebugger(mDebugger);
299 }
300
301 {
302 HelperScopedTimer timerBuild(&mStatistic, POLYTOPE_BUILD);
303 if (!createInitialPolygons(vertices0, numVertices0, vertices1, numVertices1, mConfiguration.hyperSphereNormalization))
304 {
305 std::cout << "Polygon initialization error" << std::endl;
306 return FAILURE;
307 }
308 }
309
310 if (mQueryPolygon[0]->getVertexCount() == 1 && mQueryPolygon[1]->getVertexCount() == 1)
311 {
312 std::set<SilhouetteMeshFace*> intersectedFaces;
313 result = findSceneIntersection(mQueryPolygon[0]->getVertex(0), mQueryPolygon[1]->getVertex(0), intersectedFaces) ? HIDDEN : VISIBLE;
314 }
315 else
316 {
317 {
318 HelperScopedTimer timer(getStatistic(), SILHOUETTE_PROCESSING);
319 //mScene->get()->restoreFacesGeometry(mScene);
320
321 mSilhouetteProcessor->init(*mQueryPolygon[0], *mQueryPolygon[1]);
322 extractAllSilhouettes();
323 }
324 {
325 HelperScopedTimer timer(getStatistic(), RAY_INTERSECTION);
326 mSilhouetteContainer->prepare();
327 }
328
329 {
330 HelperScopedTimer timerBuild(&mStatistic, POLYTOPE_BUILD);
331
332 PluckerPolytopeBuilder<P, S> builder(mConfiguration.hyperSphereNormalization, mTolerance);
333 PluckerPolytope<P>* myPolytope = builder.build(*mQueryPolygon[0], *mQueryPolygon[1], getComplex()->getPolyhedron());
334 getComplex()->setRoot(myPolytope);
335 }
336 VisibilitySolver<P, S>* solver;
337
338 solver = new VisibilityApertureFinder<P, S>(this, mConfiguration.hyperSphereNormalization, mTolerance, mConfiguration.detectApertureOnly);
339
340 if (mDebugger != nullptr)
341 {
342 solver->attachVisualisationDebugger(mDebugger);
343 }
344 result = solver->resolve();
345
346 delete solver;
347 }
348
349 return result;
350 }

360 {
361 HelperScopedTimer timer(getStatistic(), OCCLUDER_TREATMENT);
362#ifdef OUTPUT_DEBUG_FILE
363 std::ofstream& debugOutput = mDebugger->getDebugOutput();
364 V_LOG(debugOutput, "VisibilityExactQuery<P, S>::findTheBestValidEdge BEGIN", occlusionTreeNodeSymbol);
365#endif
366
367 const std::unordered_set<Silhouette*>& mySilhouettes = mSilhouetteContainer->getSilhouettes();
368
369 double myScore = 1e32;
370 bool found = false;
371
372 Silhouette* mySilhouette = nullptr;
373
374 MathPlane3d aPlane0 = getQueryPolygon(0)->getPlane();
375
376 const MathPlane3d& myPlane = getQueryPolygon(0)->getPlane();
377 for (auto iter = mySilhouettes.begin(); iter != mySilhouettes.end(); iter++)
378 {
379 Silhouette* s = (*iter);
380
381 auto& edges = s->getEdges();
382
383 for (size_t silhouetteEdgeIndex = 0; silhouetteEdgeIndex < edges.size(); silhouetteEdgeIndex++)
384 {
385 SilhouetteEdge& edge = edges[silhouetteEdgeIndex];
386 if (edge.mIsActive)
387 {
388 /*
389 if (edge.mScore < 0.0)
390 {
391 ConnectedMeshElement* f = edge.mFace;
392 size_t edgeIndex = edge.mEdgeIndex;
393 MathVector2i ei = f->getEdge(edgeIndex);
394 edge.mScore = aPlane0.dot(convert<MathVector3d>(f->getVertex(ei.x)));
395 }
396 if (edge.mScore < myScore)
397 */
398 {
399 found = true;
400 myScore = edge.mScore;
401 aSilhouetteEdgeIndex = silhouetteEdgeIndex;
402 mySilhouette = s;
403 }
404 }
405 if (found)
406 break;
407 }
408
409 if (found)
410 break;
411 }
412
413 if (found)
414 {
415#ifdef OUTPUT_DEBUG_FILE
416 std::stringstream ss;
417 ss << "VisibilityExactQuery<P, S>::findTheBestValidEdge END return True (Edge found)";
418 V_LOG(debugOutput, ss.str(), occlusionTreeNodeSymbol);
419#endif
420 aSilhouette = mySilhouette;
421
422 }
423 else
424 {
425#ifdef OUTPUT_DEBUG_FILE
426 V_LOG(debugOutput, "VisibilityExactQuery<P, S>::findTheBestValidEdge END return False (No edge found)", occlusionTreeNodeSymbol);
427#endif
428 }
429 return found;
430 }

438 template<class P, class S>
439 bool VisibilityExactQuery_<P, S>::findSceneIntersection(const MathVector3d & aBegin, const MathVector3d & anEnd, std::set<SilhouetteMeshFace*> & anIntersectedFaces, PluckerPolytope<P> * aPolytope)
440 {
441 MathVector3d myBegin = aBegin;
442 MathVector3d myDir = anEnd - aBegin;
443 double myMax = myDir.normalize();
444
445 VisibilityRay myRay;
446 myRay.org[0] = (float)myBegin.x;
447 myRay.org[1] = (float)myBegin.y;
448 myRay.org[2] = (float)myBegin.z;
449 myRay.dir[0] = (float)myDir.x;
450 myRay.dir[1] = (float)myDir.y;
451 myRay.dir[2] = (float)myDir.z;
452 myRay.tfar = (float)myMax;
453 myRay.tnear = 0;
454
455 if (mDebugger != nullptr)
456 {
457 mDebugger->addSamplingLine(convert<MathVector3f>(aBegin), convert<MathVector3f>(anEnd));
458 }
459
460 bool intersect = false;
461
462 {
463 HelperScopedTimer timer(getStatistic(), RAY_INTERSECTION);
464 getStatistic()->inc(RAY_COUNT);
465
466 intersect = mSilhouetteContainer->intersect(&myRay);
467 }
468
469 if (intersect)
470 {
471 for (size_t i = 0; i < myRay.mPrimitiveIds.size(); i++)
472 {
473 //:TODO: replace wihth accessors
474 size_t myPrimitive = myRay.mPrimitiveIds[i];
475 size_t geometryId = myRay.mGeometryIds[i];
476
477 std::vector<SilhouetteMeshFace>* myFaces = mScene->getOccluderConnectedFaces(geometryId);
478 SilhouetteMeshFace* myFace = &(*myFaces)[myPrimitive];
479 anIntersectedFaces.insert(myFace);
480 }
481 return true;
482 }
483
484 if (mDebugger != nullptr)
485 {
486 mDebugger->addStabbingLine(convert<MathVector3f>(aBegin), convert<MathVector3f>(anEnd));
487 }
488
489 return false;
490 }

511 bool VisibilityExactQuery_<P, S>::collectAllOccluders(PluckerPolytope<P> * aPolytope, PluckerPolyhedron<P> * polyhedron, std::vector<Silhouette*> & occluders, std::vector<P> & polytopeLines)
512 {
513 {
514 if (mConfiguration.representativeLineSampling)
515 {
516 HelperScopedTimer timer(getStatistic(), STABBING_LINE_EXTRACTION);
517
518 P myRepresentativeLine = MathGeometry::computeRepresentativeLine<P>(aPolytope, polyhedron, mTolerance);
519 if (mConfiguration.hyperSphereNormalization)
520 myRepresentativeLine = myRepresentativeLine.getNormalized();
521 polytopeLines.push_back(myRepresentativeLine);
522
523 aPolytope->setRepresentativeLine(myRepresentativeLine);
524 }
525 else
526 {
527 HelperScopedTimer timer(getStatistic(), STABBING_LINE_EXTRACTION);
528
529 if (aPolytope->getExtremalStabbingLinesCount() == 0)
530 {
531 aPolytope->computeExtremalStabbingLines(polyhedron, mTolerance);
532 }
533 for (size_t i = 0; i < aPolytope->getExtremalStabbingLinesCount(); i++)
534 {
535 polytopeLines.push_back(aPolytope->getExtremalStabbingLine(i));
536 }
537 }
538 }
539
540 const MathPlane3d& aPlane0 = getQueryPolygon(0)->getPlane();
541 const MathPlane3d& aPlane1 = getQueryPolygon(1)->getPlane();
542
543 for (size_t i = 0; i < polytopeLines.size(); i++)
544 {
545 std::pair<MathVector3d, MathVector3d> line = MathGeometry::getBackTo3D(polytopeLines[i], aPlane0, aPlane1);
546 // V_ASSERT(MathGeometry::isPointInsidePolygon(*getQueryPolygon(0), lines[i].first, MathArithmetic<double>::Tolerance()));
547 // V_ASSERT(MathGeometry::isPointInsidePolygon(*getQueryPolygon(1), lines[i].second, MathArithmetic<double>::Tolerance()));
548 std::set<SilhouetteMeshFace*> intersectedFaces;
549 bool hit = findSceneIntersection(line.first, line.second, intersectedFaces, aPolytope);
550 // if (mConfiguration.useEmbree )
551 // hit = hit || findSceneIntersection(line.second, line.first, intersectedFaces, aPolytope);
552
553 if (!hit)
554 {
555 return false;
556 }
557
558 for (auto myFace : intersectedFaces)
559 {
560 Silhouette* s = mSilhouetteProcessor->findSilhouette(myFace);
561 // V_ASSERT(s);
562 if (s)
563 occluders.push_back(s);
564 }
565 }
566 return true;
