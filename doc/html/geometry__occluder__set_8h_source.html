141 inline void GeometryOccluderSet::extractConnectedMeshFaces(GeometryDiscreteMeshDescription* mesh, std::vector<SilhouetteMeshFace>& aFaces)
142 {
143 size_t myFaceNumber = mesh->faceCount;
144
145 aFaces.clear();
146 aFaces.resize(myFaceNumber);
147
148 for (size_t i = 0; i < myFaceNumber; i++)
149 {
150 aFaces[i].setGeometry(mesh, i);
151 }
152 std::unordered_map<size_t, MathVector2i> myEdgesMap;
153
154 for (size_t myTriangle = 0; myTriangle < myFaceNumber; myTriangle++)
155 {
156 const std::vector<int> myIndices = mesh->getIndices(myTriangle);
157
158 for (size_t myEdge = 0; myEdge < 3; myEdge++)
159 {
160 size_t myBegin = myIndices[myEdge];
161 size_t myEnd = myIndices[(myEdge + 1) % myIndices.size()];
162
163 const size_t key = getKey(myBegin, myEnd, mesh->faceCount*3);
164 auto iter = myEdgesMap.find(key);
165 if (iter == myEdgesMap.end())
166 {
167 aFaces[myTriangle].setNeighbour(myEdge, -1);
168 myEdgesMap.insert(std::pair<size_t, MathVector2i>(key, MathVector2i(myTriangle, myEdge)));
169 }
170 else
171 {
172 size_t myNeighborTriangle = iter->second.x;
173 size_t myNeighborEdge = iter->second.y;
174
175 myEdgesMap.erase(iter);
176
177 aFaces[myTriangle].setNeighbour(myEdge, (int)myNeighborTriangle);
178 aFaces[myNeighborTriangle].setNeighbour(myNeighborEdge, (int)myTriangle);
179 }
180 }
181 }
182 }
