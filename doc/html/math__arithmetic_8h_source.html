169  template<class S>
170  inline void MathArithmetic<S>::getMinMax(const S* aPoints, size_t aPointCount, MathVector3_<S>& aMin, MathVector3_<S>& aMax)
171  {
172  const MathVector3_<S>* p = reinterpret_cast<const MathVector3_<S>*>(aPoints);
173  aMin = MathVector3_<S>( FLT_MAX, FLT_MAX, FLT_MAX);
174  aMax = MathVector3_<S>(-FLT_MAX, -FLT_MAX, -FLT_MAX);
175 
176  for (size_t i = 0; i < aPointCount; i++)
177  {
178  if (p[i].x < aMin.x) aMin.x = p[i].x;
179  if (p[i].y < aMin.y) aMin.y = p[i].y;
180  if (p[i].z < aMin.z) aMin.z = p[i].z;
181 
182  if (p[i].x > aMax.x) aMax.x = p[i].x;
183  if (p[i].y > aMax.y) aMax.y = p[i].y;
184  if (p[i].z > aMax.z) aMax.z = p[i].z;
185  }
186  }
