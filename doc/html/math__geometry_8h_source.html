168  inline bool MathGeometry::isPointInsidePolygon(const GeometryConvexPolygon& aPolygon, const MathVector3d& aPoint, double tolerance)
169  {
170  if (aPolygon.getVertexCount() == 0)
171  return false;
172 
173  if (aPolygon.getVertexCount() == 1)
174  {
175  MathVector3d r = aPoint - aPolygon.getVertex(0);
176  return r.getSquaredNorm() < tolerance;
177  }
178  if (aPolygon.getVertexCount() == 2)
179  {
180  MathVector3d ab = aPolygon.getVertex(0) - aPolygon.getVertex(1);
181  MathVector3d ac = aPolygon.getVertex(0) - aPoint;
182  MathVector3d bc = aPolygon.getVertex(1) - aPoint;
183 
184  double test = MathArithmetic<double>::getAbs(ac.getNorm() + bc.getNorm() - ab.getNorm());
185  //std::cout << "Test 2D: " << test << std::endl;
186  return test < 1e-6; //TODO
187  }
188  if (MathArithmetic<double>::getAbs(aPolygon.getPlane().dot(aPoint)) > tolerance)
189  return false;
190 
191  for (int i = 0; i < aPolygon.getVertexCount(); i++)
192  {
193  MathVector2i myEdge = aPolygon.getEdge(i);
194  MathVector3d myNormal = MathVector3d::cross(aPolygon.getPlane().mNormal, aPolygon.getVertex(myEdge.x) - aPolygon.getVertex(myEdge.y));
195  MathPlane3d myPlane(myNormal.x, myNormal.y, myNormal.z, -myNormal.dot(aPolygon.getVertex(i)));
196  if (myPlane.dot(aPoint) > tolerance)
197  {
198  return false;
199  }
200  }
201 
202  return true;
203  }

205  inline bool MathGeometry::isEdgePotentiallyInsideShaft(const std::vector<MathPlane3d> & aShaftPlanes, const MathVector3f & a, const MathVector3f & b, bool exclude)
206  {
207  V_ASSERT(aShaftPlanes.size() > 0);
208 
209  //Approximate using a bounding sphere
210 
211  MathVector3f myCenter = a;
212  myCenter += b;
213  myCenter *= 0.5;
214 
215  MathVector3f radiusVector = b;
216  radiusVector -= a;
217  double radius = radiusVector.getNorm()*0.5;
218 
219  MathVector3d myCenterD = convert<MathVector3d>(myCenter);
220  for (size_t i = 0; i < aShaftPlanes.size(); i++)
221  {
222  double d = aShaftPlanes[i].dot(myCenterD);
223 
224  if (exclude)
225  {
226  if (d < -radius)
227  {
228  return false;
229  }
230  }
231  else
232  {
233  if (d <= -radius)
234  {
235  return false;
236  }
237  }
238  }
239  return true;

288  }

294  template<class P, class S>
295  inline P MathGeometry::interpolate(S anOffset1, S anOffset2, const P & aVertice1, const P & aVertice2, S tolerance)
296  {
297  P myVector1 = aVertice1;
298  P myVector2 = aVertice2;
299 
300  S lambdaV1 = MathArithmetic<S>::getAbs(anOffset1);
301  S lambdaV2 = MathArithmetic<S>::getAbs(anOffset2);
302 
303  S eta = lambdaV2 - lambdaV1;
304  S gamma;
305 
306  S sum;
307  S sum2;
308 
309  if (eta < -tolerance)
310  {
311  eta = -eta;
312  sum = lambdaV2;
313  gamma = lambdaV2;
314  sum2 = lambdaV2;
315  sum += eta;
316  }
317 
318  else
319  {
320  sum = lambdaV1;
321  gamma = lambdaV1;
322  sum2 = lambdaV1;
323  myVector1 = aVertice2;
324  sum += eta;
325  myVector2 = aVertice1;
326  }
327 
328  S alpha, beta;
329 
330  sum2 += sum;
331 
332  if (eta >= tolerance || gamma >= tolerance)
333  {
334  V_ASSERT(MathArithmetic<S>::getAbs(sum2) > tolerance);
335  S div = 1.f / (sum2);
336  alpha = gamma * div;
337  beta = sum * div;
338  }
339 
340  else
341  {
342  alpha = beta = 0.5f;
343  }
344 
345  myVector1 *= alpha;
346  myVector2 *= beta;
347  myVector1 += myVector2;
348 
349  return myVector1;
350  }

379  template<class P, class S>
380  inline bool MathGeometry::findPluckerEdgeWithQuadricIntersection(const P & v1, const P & v2, GeometryPositionType p1, GeometryPositionType p2, std::vector<P> & result, bool newtonRaphson, S tolerance)
381  {
382  result.clear();
383 
384  if (p1 == ON_BOUNDARY)
385  {
386  result.push_back(v1);
387  }
388  if (p2 == ON_BOUNDARY)
389  {
390  result.push_back(v2);
391  }
392  if (result.size() > 0)
393  {
394  return true;
395  }
396  P F = v2 - v1;
397 
398  S a = F.dot(F);
399  S b = F.dot(v1);
400  S c = v1.dot(v1);
401 
402  std::vector<S> ts;
403 
404  if (a <= tolerance && a >= -tolerance)
405  {//a==0
406  if (b <= tolerance && b >= -tolerance)
407  {//b==0
408  return false;
409  }
410 
411  ts.push_back(-c / (2 * b));
412  }
413  else
414  {
415  S b_2 = b * b;
416  if (a * a + b_2 + c * c <= tolerance)
417  {
418  return false;
419  }
420  S delta = b_2 - a * c;
421  if (delta <= -tolerance)
422  {
423  return false; // delta < 0: no root
424  }
425  if (delta <= tolerance)
426  {
427  // delta is 0, single root
428  ts.push_back(-b / a);
429  }
430  else
431  {
432  //general case: delta is positive, two roots
433  S sqrt_delta = MathArithmetic<S>::getSqrt(delta);
434  V_ASSERT(MathArithmetic<S>::getAbs(a) >= tolerance);
435 
436  ts.push_back((-b + sqrt_delta) / a);
437  ts.push_back((-b - sqrt_delta) / a);
438  }
439  }
440  for (size_t i = 0; i < ts.size(); i++)
441  {
442  S t = ts[i];
443 
444  if (t >= -tolerance && t <= 1 + tolerance)
445  {
446  if (newtonRaphson)
447  {
448  S at, b2;
449  b2 = 2 * b;
450  S fxn = (a * t * t + b2 * t + c);
451 
452  int iter = 0;
453 
454  // Newton - Raphson iteration scheme : xn + 1 = xn - f(xn) / f'(xn)
455  //f (t) = a t*t + 2 b t + c
456  //f'(t) = 2 at + 2b
457  const double toleranceLocal = 1e-18;
458  while (MathArithmetic<S>::getAbs(fxn) >= toleranceLocal && iter < 20)
459  {
460  at = a * t;
461  fxn = (at * t + b2 * t + c);
462  S fdxn = (2 * at + b2);
463  t = t - fxn / fdxn;
464  iter++;
465  }
466  }
467  if (t >= -tolerance && t <= 1 + tolerance)
468  {
469  result.push_back((F * t + v1));
470  }
471  }
472  }
473 
474 #if _DEBUG
475  for (size_t i = 0; i < result.size(); i++)
476  {
477  V_ASSERT(result[i].dot(result[i]) < tolerance);
478  }
479 #endif
480  return result.size() > 0;
481  }

483  template<class P, class S>
484  inline bool MathGeometry::hasPluckerEdgeWithQuadricIntersection(const P & v1, const P & v2, GeometryPositionType p1, GeometryPositionType p2, S tolerance)
485  {
486  if (p1 == ON_BOUNDARY || p2 == ON_BOUNDARY)
487  {
488  return true;
489  }
490 
491  P F = v2 - v1;
492 
493  S a = F.dot(F);
494  S b = F.dot(v1);
495  S c = v1.dot(v1);
496 
497  std::vector<S> ts;
498 
499  if (a <= tolerance && a >= -tolerance)
500  {
501  if (b <= tolerance && b >= -tolerance)
502  {
503  return false;
504  }
505 
506  S t = (-c / (2 * b));
507  return (t >= -tolerance && t <= 1 + tolerance);
508  }
509  else
510  {
511  S b_2 = b * b;
512  if (a * a + b_2 + c * c <= tolerance)
513  {
514  return false;
515  }
516  S delta = b_2 - a * c;
517  if (delta <= -tolerance)
518  {
519  return false; // delta < 0: no root
520  }
521  if (delta <= tolerance)
522  {
523  // delta is 0, single root
524  S t = (-b / a);
525  return (t >= -tolerance && t <= 1 + tolerance);
526  }
527  else
528  {
529  //general case: delta is positive, two roots
530  S sqrt_delta = MathArithmetic<S>::getSqrt(delta);
531  V_ASSERT(MathArithmetic<S>::getAbs(a) >= tolerance);
532 
533  S t = (-b + sqrt_delta) / a;
534  if (t >= -tolerance && t <= 1 + tolerance)
535  return true;
536  t = (-b - sqrt_delta) / a;
537  if (t >= -tolerance && t <= 1 + tolerance)
538  return true;
539 
540  return false;
541  }
542  }
543 
544  return false;
545  }

553  inline bool MathGeometry::clipWithGardBand(GeometryConvexPolygon & aPolygon, const MathPlane3d & aPlane, double anEpsilon)
554  {
555  MathPlane3d myPlane(aPlane);
556  myPlane.d -= anEpsilon;
557 
558  enum
559  {
560  polygonInterior = 1,
561  polygonBoundary = 0,
562  polygonExterior = -1
563  };
564 
565  std::vector<signed char> location(aPolygon.getVertexCount());
566 
567  int positive = 0;
568  int negative = 0;
569 
570  for (size_t a = 0; a < aPolygon.getVertexCount(); a++)
571  {
572  double d = myPlane.dot(aPolygon.getVertex(a));
573  if (d > MathArithmetic<double>::Tolerance())
574  {
575  location[a] = polygonInterior;
576  positive++;
577  }
578  else if (d < -MathArithmetic<double>::Tolerance())
579  {
580  location[a] = polygonExterior;
581  negative++;
582  }
583  else
584  {
585  location[a] = polygonBoundary;
586  }
587  }
588 
589  if (negative == 0)
590  {
591  return true;
592  }
593  else if (positive == 0)
594  {
595  aPolygon.clear();
596  return false;
597  }
598  // std::cout<<"CLIP -the query polygon"<<std::endl;
599  size_t previous = aPolygon.getVertexCount() - 1;
600 
601  std::vector<MathVector3d> result;
602 
603  for (int index = 0; index < aPolygon.getVertexCount(); index++)
604  {
605  long loc = location[index];
606  if (loc == polygonExterior)
607  {
608  if (location[previous] == polygonInterior)
609  {
610  const MathVector3d& v1 = aPolygon.getVertex(previous);
611  const MathVector3d& v2 = aPolygon.getVertex(index);
612  MathVector3d myRes = getPlaneIntersectionWithEdge(v1, v2, myPlane);
613  result.push_back(myRes);
614  }
615  }
616  else
617  {
618  const MathVector3d& v1 = aPolygon.getVertex(index);
619  if ((loc == polygonInterior) && (location[previous] == polygonExterior))
620  {
621  const MathVector3d& v2 = aPolygon.getVertex(previous);
622  MathVector3d myRes = getPlaneIntersectionWithEdge(v1, v2, myPlane);
623  result.push_back(myRes);
624  }
625 
626  result.push_back(v1);
627  }
628 
629  previous = index;
630  }
631  //TODO TEST! (not tested)
632  aPolygon = GeometryConvexPolygon(result);
633 
634  V_ASSERT(aPolygon.getVertexCount() > 1);
635  return (aPolygon.getVertexCount() > 1);
636  }

638  inline MathPlane3d MathGeometry::computePlane(const GeometryConvexPolygon & polygon)
639  {
640  bool edgeNotFound = true;
641  size_t i = 1;
642  MathVector3d myNormal;
643 
644  while (edgeNotFound && i < polygon.getVertexCount() - 1)
645  {
646  myNormal = MathVector3d::cross(polygon.getVertex(i) - polygon.getVertex(i - 1), polygon.getVertex(i + 1) - polygon.getVertex(i));
647 
648  edgeNotFound = (myNormal.getSquaredNorm() < MathArithmetic<double>::Tolerance());
649  i++;
650  }
651 
652  MathVector3d axes[3] = { MathVector3d::X(), MathVector3d::Y(), MathVector3d::Z() };
653 
654  if (edgeNotFound && polygon.getVertexCount() > 1)
655  {
656  MathVector3d myEdge1 = polygon.getVertex(0) - polygon.getVertex(1);
657 
658  while (edgeNotFound && i < 3)
659  {
660  myNormal = MathVector3d::cross(myEdge1, axes[i]);
661  edgeNotFound = (myNormal.getSquaredNorm() < MathArithmetic<double>::Tolerance());
662  i++;
663  }
664  }
665  if (edgeNotFound)
666  {
667  myNormal = axes[0];
668  edgeNotFound = false;
669  }
670  V_ASSERT(!edgeNotFound);
671  myNormal.normalize();
672 
673  return MathPlane3d(myNormal.x, myNormal.y, myNormal.z, -myNormal.dot(polygon.getVertex(0)));
674  }

700  inline std::pair<MathVector3d, MathVector3d> MathGeometry::getBackTo3D(const P & aPoint, const MathPlane3d & aPlane1, const MathPlane3d & aPlane2)
701  {
702  MathVector3d myBegin = MathGeometry::getAnchorPoint(aPoint);
703  MathVector3d myDir = convert<MathVector3d>(aPoint.getDirection());
704  V_ASSERT(!myDir.isZero(MathArithmetic<double>::Tolerance()));
705  double myMax = myDir.normalize();
706  V_ASSERT(myMax > MathArithmetic<double>::Tolerance());
707  MathVector3d myResult1;
708  MathVector3d myResult2;
709  bool result1 = getPlaneIntersection(aPlane1, myBegin, myDir, myResult1, MathArithmetic<double>::Tolerance());
710  bool result2 = getPlaneIntersection(aPlane2, myBegin, myDir, myResult2, MathArithmetic<double>::Tolerance());
711  V_ASSERT(result1 && result2);
712 
713  return std::pair<MathVector3d, MathVector3d>(myResult1, myResult2);
714  }
